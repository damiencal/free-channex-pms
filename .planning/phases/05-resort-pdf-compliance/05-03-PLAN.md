"""Async email delivery for resort booking form submissions.

Uses aiosmtplib for async SMTP and tenacity for exponential backoff retry.
Supports both TLS (port 465) and STARTTLS (port 587) based on port number.

Retry strategy: 4 total attempts with exponential backoff (10s, 20s, 40s max 120s).
Only surfaces error after all retries are exhausted.
"""

Implement two public functions:

**`send_resort_email(...)` — Core send function:**

```python
from email.message import EmailMessage
import aiosmtplib
import structlog

log = structlog.get_logger()

async def send_resort_email(
    smtp_host: str,
    smtp_port: int,
    smtp_user: str,
    smtp_password: str,
    from_addr: str,
    to_addr: str,
    subject: str,
    body: str,
    filled_form_bytes: bytes,
    filled_form_filename: str,
    confirmation_bytes: bytes | None = None,
    confirmation_filename: str | None = None,
) -> None:
    """Send resort form email with PDF attachments.

    Always attaches the filled booking form. Optionally attaches the platform
    booking confirmation if available (confirmation_bytes not None).

    Args:
        smtp_host: SMTP server hostname.
        smtp_port: SMTP port (465 for TLS, 587 for STARTTLS).
        smtp_user: SMTP username for authentication.
        smtp_password: SMTP password.
        from_addr: Sender email address.
        to_addr: Recipient email address (resort contact).
        subject: Email subject line.
        body: Email body text (plain text).
        filled_form_bytes: Filled PDF form as bytes.
        filled_form_filename: Filename for the form attachment (e.g., "BookingForm_JohnSmith_Lot110.pdf").
        confirmation_bytes: Platform booking confirmation PDF as bytes, or None if not available.
        confirmation_filename: Filename for the confirmation attachment, or None.

    Raises:
        aiosmtplib.SMTPException: On SMTP delivery failure.
    """
    msg = EmailMessage()
    msg["From"] = from_addr
    msg["To"] = to_addr
    msg["Subject"] = subject
    msg.set_content(body)

    # Always attach the filled booking form
    msg.add_attachment(
        filled_form_bytes,
        maintype="application",
        subtype="pdf",
        filename=filled_form_filename,
    )

    # Optionally attach the platform booking confirmation
    if confirmation_bytes is not None and confirmation_filename is not None:
        msg.add_attachment(
            confirmation_bytes,
            maintype="application",
            subtype="pdf",
            filename=confirmation_filename,
        )

    await aiosmtplib.send(
        msg,
        hostname=smtp_host,
        port=smtp_port,
        username=smtp_user,
        password=smtp_password,
        use_tls=(smtp_port == 465),   # Direct TLS for port 465
        start_tls=(smtp_port == 587), # STARTTLS for port 587
    )

    log.info(
        "Resort email sent",
        to=to_addr,
        subject=subject,
        attachments=2 if confirmation_bytes else 1,
    )
```

**`send_with_retry(...)` — Retry wrapper:**

```python
from tenacity import (
    retry,
    stop_after_attempt,
    wait_exponential,
    before_sleep_log,
    RetryError,
)
import logging

# Use stdlib logging for tenacity's before_sleep_log (it expects stdlib, not structlog)
_retry_logger = logging.getLogger("app.compliance.emailer.retry")

@retry(
    stop=stop_after_attempt(4),
    wait=wait_exponential(multiplier=2, min=10, max=120),
    before_sleep=before_sleep_log(_retry_logger, logging.WARNING),
    reraise=True,
)
async def send_with_retry(
    smtp_host: str,
    smtp_port: int,
    smtp_user: str,
    smtp_password: str,
    from_addr: str,
    to_addr: str,
    subject: str,
    body: str,
    filled_form_bytes: bytes,
    filled_form_filename: str,
    confirmation_bytes: bytes | None = None,
    confirmation_filename: str | None = None,
) -> None:
    """Send resort email with exponential backoff retry.

    Wraps send_resort_email() with tenacity retry logic:
    - 4 total attempts (1 initial + 3 retries)
    - Exponential backoff: 10s, 20s, 40s (capped at 120s)
    - Logs each retry at WARNING level
    - Re-raises the final exception after all attempts fail

    Args:
        Same as send_resort_email().

    Raises:
        aiosmtplib.SMTPException: After all 4 attempts fail (re-raised by tenacity).
    """
    await send_resort_email(
        smtp_host=smtp_host,
        smtp_port=smtp_port,
        smtp_user=smtp_user,
        smtp_password=smtp_password,
        from_addr=from_addr,
        to_addr=to_addr,
        subject=subject,
        body=body,
        filled_form_bytes=filled_form_bytes,
        filled_form_filename=filled_form_filename,
        confirmation_bytes=confirmation_bytes,
        confirmation_filename=confirmation_filename,
    )
```

Note: `reraise=True` means tenacity re-raises the original exception (not RetryError) after exhausting all attempts. The caller (submission orchestrator) catches this to record the error in ResortSubmission.error_message.

IMPORTANT:
- Use Python stdlib `email.message.EmailMessage` for MIME construction — NOT string building.
- confirmation_bytes is Optional — when None, only the filled form is attached. This handles the "confirmation file not yet available" case gracefully.
- Use `use_tls` for port 465 and `start_tls` for port 587 — supports both SMTP configurations.

**`find_confirmation_file(confirmation_code, confirmations_dir) -> Path | None`:**

```python
from pathlib import Path
import structlog

log = structlog.get_logger()

def find_confirmation_file(
    confirmation_code: str,
    confirmations_dir: str,
) -> Path | None:
    """Find a platform booking confirmation PDF by confirmation code.

    Scans the confirmations directory for PDF files whose name contains
    the confirmation code (case-insensitive). Returns the first match.

    Confirmation files are placed in this directory by n8n or a mail rule
    automation. Filenames contain the confirmation code, e.g., "HMAB1234.pdf"
    or "booking_HMAB1234_confirmation.pdf".

    Args:
        confirmation_code: Platform booking ID / confirmation code to search for.
        confirmations_dir: Path to the directory containing confirmation PDFs.

    Returns:
        Path to the matching PDF file, or None if not found.
    """
    base = Path(confirmations_dir)
    if not base.exists():
        log.warning(
            "Confirmations directory does not exist",
            dir=confirmations_dir,
        )
        return None

    code_lower = confirmation_code.lower()
    for pdf_file in sorted(base.glob("*.pdf")):
        if code_lower in pdf_file.name.lower():
            log.info(
                "Confirmation file found",
                code=confirmation_code,
                file=pdf_file.name,
            )
            return pdf_file

    log.warning(
        "No confirmation file found",
        code=confirmation_code,
        dir=confirmations_dir,
    )
    return None
```

```python
from datetime import date

def format_email_subject(
    guest_name: str,
    lot_number: str,
    check_in: date,
    check_out: date,
) -> str:
    """Format the email subject line per locked decision from CONTEXT.md.

    Format: "Booking Form - {Guest Name} - Lot {number} - {dates}"
    Same-month: "Mar 5-8"
    Cross-month: "Mar 31 - Apr 2"

    Args:
        guest_name: Full guest name.
        lot_number: Resort lot/site number (e.g., "110").
        check_in: Check-in date.
        check_out: Check-out date.

    Returns:
        Formatted subject line string.

    Examples:
        >>> format_email_subject("John Smith", "110", date(2026, 3, 5), date(2026, 3, 8))
        'Booking Form - John Smith - Lot 110 - Mar 5-8'
        >>> format_email_subject("Jane Doe", "170", date(2026, 3, 31), date(2026, 4, 2))
        'Booking Form - Jane Doe - Lot 170 - Mar 31 - Apr 2'
    """
    if check_in.month == check_out.month:
        dates_str = f"{check_in.strftime('%b')} {check_in.day}-{check_out.day}"
    else:
        dates_str = (
            f"{check_in.strftime('%b')} {check_in.day} - "
            f"{check_out.strftime('%b')} {check_out.day}"
        )
    return f"Booking Form - {guest_name} - Lot {lot_number} - {dates_str}"
```

def format_email_body(contact_name: str, sender_name: str = "Thomas") -> str:
