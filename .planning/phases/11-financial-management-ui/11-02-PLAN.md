Create fetch functions:

- `fetchTransactions(filters: TransactionFilters): Promise<BankTransactionResponse[]>` — builds query string from filters, calls GET /accounting/bank-transactions
- `categorizeTransaction(txnId: number, body: SingleCategoryRequest): Promise<BankTransactionResponse>` — PATCH /accounting/bank-transactions/{txnId}/category
- `bulkCategorize(assignments: CategoryAssignment[]): Promise<{categorized: number, errors: unknown[]}>` — PATCH /accounting/bank-transactions/categorize with `{assignments}` body
- `createExpense(body: ExpenseRequest): Promise<ExpenseResponse>` — POST /accounting/expenses
- `fetchLoans(): Promise<LoanResponse[]>` — GET /accounting/loans
- `recordLoanPayment(body: LoanPaymentRequest): Promise<LoanPaymentResult>` — POST /accounting/loans/payments
- `fetchUnreconciled(propertyId?: number | null): Promise<UnreconciledResponse>` — GET /accounting/reconciliation/unreconciled with optional property_id param
- `runReconciliation(): Promise<ReconciliationRunResult>` — POST /accounting/reconciliation/run
- `confirmMatch(body: MatchConfirmRequest): Promise<{status: string}>` — POST /accounting/reconciliation/confirm
- `rejectMatch(matchId: number): Promise<{status: string}>` — POST /accounting/reconciliation/reject/{matchId}
- `fetchFinanceSummary(propertyId?: number | null): Promise<FinanceSummaryResponse>` — GET /accounting/finance-summary with optional property_id param

For query string building, construct URLSearchParams and append to the path. Example:
```typescript
export function fetchTransactions(filters: TransactionFilters = {}): Promise<BankTransactionResponse[]> {
  const params = new URLSearchParams()
  if (filters.categorized) params.set('categorized', filters.categorized)
  if (filters.start_date) params.set('start_date', filters.start_date)
  // ... etc
  const qs = params.toString()
  return apiFetch<BankTransactionResponse[]>(`/accounting/bank-transactions${qs ? `?${qs}` : ''}`)
}
```

**FinanceTab.tsx:**

Create the top-level Finance tab component with three sub-tabs. Use the existing `Tabs`, `TabsList`, `TabsTrigger`, `TabsContent` from `@/components/ui/tabs`.

Sub-tab URL sync: Read/write `ftab` search param alongside the existing `tab` param. Default sub-tab is `transactions`.

```typescript
import { useSearchParams } from 'react-router-dom'
import { Tabs, TabsList, TabsTrigger, TabsContent } from '@/components/ui/tabs'
// Import sub-tab components — these will be created by Plans 03-05
// Use placeholder components initially that will be replaced
import { TransactionsTab } from './TransactionsTab'
import { ExpensesLoansTab } from './ExpensesLoansTab'
import { ReconciliationTab } from './ReconciliationTab'

type FinanceSubTab = 'transactions' | 'expenses-loans' | 'reconciliation'
const VALID_SUBTABS: FinanceSubTab[] = ['transactions', 'expenses-loans', 'reconciliation']

function isValidSubTab(value: string | null): value is FinanceSubTab {
  return VALID_SUBTABS.includes(value as FinanceSubTab)
}

export function FinanceTab() {
  const [searchParams, setSearchParams] = useSearchParams()
  const rawSubTab = searchParams.get('ftab')
  const activeSubTab: FinanceSubTab = isValidSubTab(rawSubTab) ? rawSubTab : 'transactions'

  function handleSubTabChange(value: string) {
    const newParams = new URLSearchParams(searchParams)
    if (value === 'transactions') {
      newParams.delete('ftab')
    } else {
      newParams.set('ftab', value)
    }
    setSearchParams(newParams)
  }

  return (
    <div className="space-y-4">
      <Tabs value={activeSubTab} onValueChange={handleSubTabChange}>
        <div className="overflow-x-auto">
          <TabsList>
            <TabsTrigger value="transactions">Transactions</TabsTrigger>
            <TabsTrigger value="expenses-loans">Expenses & Loans</TabsTrigger>
            <TabsTrigger value="reconciliation">Reconciliation</TabsTrigger>
          </TabsList>
        </div>

        <TabsContent value="transactions">
          <TransactionsTab />
        </TabsContent>
        <TabsContent value="expenses-loans">
          <ExpensesLoansTab />
        </TabsContent>
        <TabsContent value="reconciliation">
          <ReconciliationTab />
        </TabsContent>
      </Tabs>
    </div>
  )
}
```
