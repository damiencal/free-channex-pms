    **Amount normalization (silent per CONTEXT.md):**
    ```python
    def _normalize_amount(raw: str | None, row_num: int) -> tuple[Decimal | None, str | None]:
        """Strip apostrophe, $, commas from amount string. Return (value, error_or_None)."""
        if raw is None or raw.strip() == "":
            return None, f"Row {row_num}: Amount is missing"
        cleaned = raw.strip().lstrip("'").replace("$", "").replace(",", "")
        try:
            return Decimal(cleaned), None
        except Exception:
            return None, f"Row {row_num}: Amount is not a number ({raw!r})"
    ```

    **Date parsing:**
    ```python
    def _parse_date(raw: str | None, col_name: str, row_num: int) -> tuple[date | None, str | None]:
        """Parse date string to Python date. Try common US formats."""
        if raw is None or raw.strip() == "":
            return None, f"Row {row_num}: {col_name} is missing"
        for fmt in ("%m/%d/%Y", "%m/%d/%y", "%Y-%m-%d"):
            try:
                return datetime.strptime(raw.strip(), fmt).date(), None
            except ValueError:
                continue
        return None, f"Row {row_num}: {col_name} is not a valid date ({raw!r})"
    ```
