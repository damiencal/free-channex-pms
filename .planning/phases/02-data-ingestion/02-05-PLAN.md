    **Transaction ID generation (if no native ID column):**
    ```python
    import hashlib

    def _generate_transaction_id(date_str: str, amount_str: str, description: str) -> str:
        """Generate a stable composite transaction ID when Mercury CSV has no native ID."""
        composite = f"{date_str}|{amount_str}|{description}"
        hash_hex = hashlib.sha256(composite.encode()).hexdigest()[:16]
        return f"mercury-{hash_hex}"
    ```

    **Amount parsing:**
    ```python
    def _normalize_amount(raw: str | None, row_num: int) -> tuple[Decimal | None, str | None]:
        if raw is None or raw.strip() == "":
            return None, f"Row {row_num}: Amount is missing"
        cleaned = raw.strip().replace("$", "").replace(",", "")
        try:
            return Decimal(cleaned), None
        except Exception:
            return None, f"Row {row_num}: Amount is not a number ({raw!r})"
    ```
