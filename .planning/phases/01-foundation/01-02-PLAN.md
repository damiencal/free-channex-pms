**load_all_properties() function:**
```python
def load_all_properties(config_dir: Path) -> list[PropertyConfig]:
    """Discover and validate all per-property YAML config files."""
    config_dir = Path(config_dir)
    errors: list[str] = []
    properties: list[PropertyConfig] = []

    yaml_files = sorted(config_dir.glob("*.yaml"))
    property_files = [f for f in yaml_files if f.name not in ("base.yaml", "config.example.yaml")]

    if not property_files:
        errors.append(f"{config_dir}: no property config files found (expected *.yaml)")

    slugs_seen: set[str] = set()
    for yaml_file in property_files:
        try:
            data = yaml.safe_load(yaml_file.read_text())
            if data is None:
                errors.append(f"{yaml_file.name}: file is empty")
                continue
            prop = PropertyConfig(**data)
            if prop.slug in slugs_seen:
                errors.append(f"{yaml_file.name}: duplicate slug '{prop.slug}'")
            slugs_seen.add(prop.slug)
            properties.append(prop)
        except ValidationError as e:
            for err in e.errors():
                field = " -> ".join(str(loc) for loc in err["loc"])
                errors.append(f"{yaml_file.name}: {field}: {err['msg']}")
        except Exception as e:
            errors.append(f"{yaml_file.name}: {e}")

    if errors:
        raise SystemExit(
            "Config validation failed:\n" + "\n".join(f"  - {e}" for e in errors)
        )
    return properties
```

Key details:
- Discovers *.yaml dynamically (not hardcoded to jay/minnie) â€” CONTEXT.md: "extensible for additional rental properties"
- Excludes base.yaml and config.example.yaml from property discovery
- Checks for duplicate slugs (RESEARCH.md pitfall 7)
- Formats Pydantic ValidationError into human-readable lines (RESEARCH.md open question 3)
- Calls `sys.exit()` on error (fail-fast from CONTEXT.md)

**load_app_config() function:**
```python
_config: AppConfig | None = None

def load_app_config(config_dir: str = "config") -> AppConfig:
    """Load and cache the full application config."""
    global _config
    if _config is not None:
        return _config
    config = AppConfig()
    config.properties = load_all_properties(Path(config_dir))
    _config = config
    return config

def get_config() -> AppConfig:
    """Get cached config. Raises if load_app_config() hasn't been called."""
    if _config is None:
        raise RuntimeError("Config not loaded â€” call load_app_config() first")
    return _config
```

Singleton pattern: load once at startup, access via get_config() thereafter.
