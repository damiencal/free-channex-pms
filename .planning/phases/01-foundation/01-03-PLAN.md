**app/db.py** — engine, session factory, Base:
```python
import os
from sqlalchemy import create_engine
from sqlalchemy.orm import DeclarativeBase, sessionmaker

DATABASE_URL = os.environ.get("DATABASE_URL", "postgresql+psycopg://rental:changeme@localhost:5432/rental_management")

engine = create_engine(DATABASE_URL)
SessionLocal = sessionmaker(bind=engine)

class Base(DeclarativeBase):
    pass

def get_db():
    """Dependency for FastAPI — yields a session, closes on exit."""
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()
```

Key details:
- Use `postgresql+psycopg://` (NOT `+psycopg3`) — RESEARCH.md pitfall 4
- DATABASE_URL from environment variable (loaded by Docker from .env)
- Sync engine (not async) — RESEARCH.md recommendation for Phase 1 simplicity
- `get_db()` is a FastAPI dependency generator pattern
- Default URL for local development (overridden by Docker .env in production)

**app/models/__init__.py** — import all models so Base.metadata knows about them:
```python
# Import all models here so Alembic can detect them via Base.metadata
```

This is CRITICAL for Alembic autogenerate. If models aren't imported before `target_metadata = Base.metadata`, Alembic won't see them.

**app/models/property.py** — Property ORM model using SQLAlchemy 2.0 Mapped[] style:
```python
from datetime import datetime
from sqlalchemy import String, DateTime
from sqlalchemy.orm import Mapped, mapped_column
from sqlalchemy.sql import func
from app.db import Base

class Property(Base):
    __tablename__ = "properties"

    id: Mapped[int] = mapped_column(primary_key=True)
    slug: Mapped[str] = mapped_column(String(64), unique=True, nullable=False, index=True)
    display_name: Mapped[str] = mapped_column(String(255), nullable=False)
    created_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True), server_default=func.now()
    )
    updated_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True), server_default=func.now(), onupdate=func.now()
    )

    def __repr__(self) -> str:
        return f"<Property(slug='{self.slug}', display_name='{self.display_name}')>"
```

Key details:
- Integer PK (`id`) for fast FK joins — RESEARCH.md discretion recommendation
- String `slug` as unique business key — used in config filenames, template directories, URL paths
- `created_at`/`updated_at` with timezone-aware timestamps and server defaults
- `onupdate=func.now()` for automatic updated_at tracking
- Phase 1 ONLY creates the properties table — CONTEXT.md: "Minimal per phase"
- Do NOT add booking, transaction, or journal_entry models yet — those come in Phases 2-3

```python
import os

def run_migrations_online():
    # Override alembic.ini sqlalchemy.url with environment variable
    db_url = os.environ.get("DATABASE_URL", "postgresql+psycopg://rental:changeme@localhost:5432/rental_management")
    configuration = config.get_section(config.config_ini_section, {})
    configuration["sqlalchemy.url"] = db_url

    connectable = engine_from_config(
        configuration,
        prefix="sqlalchemy.",
        poolclass=pool.NullPool,
    )
    with connectable.connect() as connection:
        context.configure(
            connection=connection,
            target_metadata=target_metadata,
            compare_type=True,
        )
        with context.begin_transaction():
            context.run_migrations()
```

4. Also handle `run_migrations_offline()` to use DATABASE_URL:

```python
def run_migrations_offline():
    url = os.environ.get("DATABASE_URL", config.get_main_option("sqlalchemy.url"))
    context.configure(
        url=url,
        target_metadata=target_metadata,
        literal_binds=True,
        dialect_opts={"paramstyle": "named"},
```

```python
"""create properties table

Revision ID: 001
Revises:
Create Date: 2026-02-27
"""
from alembic import op
import sqlalchemy as sa

revision = "001"
down_revision = None
branch_labels = None
depends_on = None

def upgrade() -> None:
    op.create_table(
        "properties",
        sa.Column("id", sa.Integer(), primary_key=True),
        sa.Column("slug", sa.String(64), nullable=False),
        sa.Column("display_name", sa.String(255), nullable=False),
        sa.Column("created_at", sa.DateTime(timezone=True), server_default=sa.func.now()),
        sa.Column("updated_at", sa.DateTime(timezone=True), server_default=sa.func.now()),
    )
    op.create_index("ix_properties_slug", "properties", ["slug"], unique=True)

def downgrade() -> None:
    op.drop_index("ix_properties_slug", table_name="properties")
    op.drop_table("properties")
```

The migration creates the table and a unique index on slug separately (cleaner than inline unique=True for migration readability).
